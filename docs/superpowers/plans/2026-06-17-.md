```python
from __future__ import annotations

import json
import shutil
from dataclasses import dataclass, field, asdict
from pathlib import Path
from typing import List

DEFAULT_CONFIG = {
    "interval_sec": 5,
    "keywords": [],
    "regions": [],
    "click_points": [],
}


@dataclass
class Region:
    name: str
    top: int
    left: int
    width: int
    height: int

    @property
    def bbox(self) -> dict:
        return {"top": self.top, "left": self.left, "width": self.width, "height": self.height}


@dataclass
class ClickPoint:
    name: str
    x: int
    y: int
    order: int = 1


@dataclass
class Config:
    interval_sec: int = 5
    keywords: List[str] = field(default_factory=list)
    regions: List[Region] = field(default_factory=list)
    click_points: List[ClickPoint] = field(default_factory=list)

    @classmethod
    def default(cls) -> "Config":
        return cls(
            interval_sec=DEFAULT_CONFIG["interval_sec"],
            keywords=list(DEFAULT_CONFIG["keywords"]),
            regions=[],
            click_points=[],
        )


def _coerce_region(d: dict) -> Region:
    b = d.get("bbox", d)
    return Region(
        name=str(d.get("name", "")),
        top=int(b.get("top", 0)),
        left=int(b.get("left", 0)),
        width=int(b.get("width", 0)),
        height=int(b.get("height", 0)),
    )


def _coerce_point(d: dict) -> ClickPoint:
    return ClickPoint(
        name=str(d.get("name", "")),
        x=int(d.get("x", 0)),
        y=int(d.get("y", 0)),
        order=int(d.get("order", 1)),
    )


def load_config(path: Path) -> Config:
    if not path.exists():
        return Config.default()
    try:
        raw = json.loads(path.read_text(encoding="utf-8"))
    except (json.JSONDecodeError, OSError):
        bak = path.with_suffix(path.suffix + ".bak")
        try:
            shutil.copy2(path, bak)
        except OSError:
            pass
        return Config.default()
    if not isinstance(raw, dict):
        return Config.default()
    return Config(
        interval_sec=int(raw.get("interval_sec", DEFAULT_CONFIG["interval_sec"])),
        keywords=[str(k) for k in raw.get("keywords", [])],
        regions=[_coerce_region(r) for r in raw.get("regions", []) if isinstance(r, dict)],
        click_points=[_coerce_point(p) for p in raw.get("click_points", []) if isinstance(p, dict)],
    )


def save_config(cfg: Config, path: Path) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    data = {
        "interval_sec": cfg.interval_sec,
        "keywords": cfg.keywords,
        "regions": [
            {"name": r.name, "bbox": {"top": r.top, "left": r.left, "width": r.width, "height": r.height}}
            for r in cfg.regions
        ],
        "click_points": [{"name": p.name, "x": p.x, "y": p.y, "order": p.order} for p in cfg.click_points],
    }
    tmp = path.with_suffix(path.suffix + ".tmp")
    tmp.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
    tmp.replace(path)
```

```python
from __future__ import annotations

from typing import Iterable, Tuple


def match_keywords(text: str, keywords: Iterable[str]) -> Tuple[bool, str]:
    """Return (matched, first_matched_keyword). Empty/stripped keywords are ignored."""
    if not text or not keywords:
        return False, ""
    for raw in keywords:
        kw = raw.strip()
        if kw and kw in text:
            return True, kw
    return False, ""
```

```python
from __future__ import annotations

import io
from typing import Mapping

import mss
from PIL import Image


class CaptureError(RuntimeError):
    pass


def capture_region(bbox: Mapping[str, int]) -> bytes:
    """Capture a screen region and return PNG bytes.

    bbox requires keys: top, left, width, height (all >= 0).
    """
    try:
        top = int(bbox["top"])
        left = int(bbox["left"])
        width = int(bbox["width"])
        height = int(bbox["height"])
    except (KeyError, TypeError, ValueError) as e:
        raise CaptureError(f"invalid bbox: {bbox}") from e
    if width <= 0 or height <= 0 or top < 0 or left < 0:
        raise CaptureError(f"invalid bbox dimensions: {bbox}")

    region = {"top": top, "left": left, "width": width, "height": height}
    try:
        with mss.mss() as sct:
            shot = sct.grab(region)
            img = Image.frombytes("RGB", shot.size, shot.bgra, "raw", "BGRX")
    except Exception as e:
        raise CaptureError(f"screenshot failed: {e}") from e

    buf = io.BytesIO()
    img.save(buf, format="PNG")
    return buf.getvalue()
```

```python
from __future__ import annotations

import csv
import re
from datetime import datetime
from pathlib import Path
from typing import Union

ILLEGAL = re.compile(r'[\\/:\*\?"<>\|\x00-\x1f]')


def sanitize_filename(name: str) -> str:
    return ILLEGAL.sub("_", name).strip() or "unnamed"


def save_screenshot(
    png_bytes: bytes,
    region_name: str,
    index: int,
    ts: datetime,
    root: Union[str, Path] = "src/pic",
) -> Path:
    root = Path(root)
    day_dir = root / ts.strftime("%Y%m%d")
    day_dir.mkdir(parents=True, exist_ok=True)
    time_part = ts.strftime("%H%M%S")
    fname = f"{time_part}_{sanitize_filename(region_name)}_{index}.png"
    path = day_dir / fname
    path.write_bytes(png_bytes)
    return path


def append_csv_row(
    ts: datetime,
    region: str,
    text: str,
    matched_keyword: str,
    root: Union[str, Path] = "src/csv",
) -> Path:
    root = Path(root)
    root.mkdir(parents=True, exist_ok=True)
    path = root / f"{ts.strftime('%Y%m%d')}.csv"
    new_file = not path.exists()
    with path.open("a", encoding="utf-8-sig", newline="") as f:
        writer = csv.writer(f)
        if new_file:
            writer.writerow(["timestamp", "region", "text", "matched_keyword"])
        writer.writerow([ts.strftime("%Y-%m-%d %H:%M:%S"), region, text, matched_keyword])
    return path
```

```python
from __future__ import annotations

import io
from typing import List

import easyocr
import numpy as np
from PIL import Image

_reader: "easyocr.Reader | None" = None


def get_reader(gpu: bool = False) -> "easyocr.Reader":
    global _reader
    if _reader is None:
        _reader = easyocr.Reader(["ch_sim", "en"], gpu=gpu, verbose=False)
    return _reader


def recognize_text(png_bytes: bytes) -> str:
    """Run OCR on PNG bytes; return concatenated recognized text (one line per detection)."""
    img = Image.open(io.BytesIO(png_bytes)).convert("RGB")
    arr = np.array(img)
    reader = get_reader()
    results: List = reader.readtext(arr, detail=0, paragraph=False)
    return "\n".join(str(r) for r in results)
```

```python
from __future__ import annotations

import time
from typing import Iterable, List, Tuple

import pyautogui

from core.config import ClickPoint


def click_points(points: Iterable[ClickPoint], interval_sec: float = 0.1) -> List[Tuple[str, bool, str]]:
    """Click each point in order. Return list of (name, success, error_msg)."""
    results: List[Tuple[str, bool, str]] = []
    for p in sorted(points, key=lambda x: x.order):
        try:
            pyautogui.moveTo(p.x, p.y, duration=0.05)
            pyautogui.click()
            results.append((p.name, True, ""))
        except Exception as e:
            results.append((p.name, False, str(e)))
        if interval_sec > 0:
            time.sleep(interval_sec)
    return results
```

```python
from __future__ import annotations

import threading
import traceback
from dataclasses import dataclass
from datetime import datetime
from typing import List, Optional

from PySide6.QtCore import QObject, QThread, QTimer, pyqtSignal

from core.capture import capture_region, CaptureError
from core.clicker import click_points
from core.config import Config
from core.matcher import match_keywords
from core.ocr import recognize_text
from core.storage import append_csv_row, save_screenshot


@dataclass
class ScanResult:
    timestamp: datetime
    region: str
    text: str
    matched_keyword: str
    image_path: str
    success: bool
    error: str = ""


class ScanWorker(QObject):
    finished_round = pyqtSignal(list)  # List[ScanResult]
    log = pyqtSignal(str)

    def __init__(self, config: Config, stop_event: threading.Event):
        super().__init__()
        self._config = config
        self._stop_event = stop_event
        self._busy = False

    def run_once(self) -> None:
        if self._busy:
            return
        self._busy = True
        try:
            results = self._do_round()
            self.finished_round.emit(results)
        finally:
            self._busy = False

    def _do_round(self) -> List[ScanResult]:
        results: List[ScanResult] = []
        ts = datetime.now()

        for idx, region in enumerate(self._config.regions, start=1):
            if self._stop_event.is_set():
                break
            try:
                png = capture_region(region.bbox)
                img_path = save_screenshot(png, region.name, idx, ts)
            except CaptureError as e:
                self.log.emit(f"[截图失败] {region.name}: {e}")
                continue
            except Exception as e:
                self.log.emit(f"[未知错误] {region.name}: {e}")
                traceback.print_exc()
                continue

            try:
                text = recognize_text(png)
            except Exception as e:
                text = f"[ERROR: {e}]"

            matched, kw = match_keywords(text, self._config.keywords)
            results.append(
                ScanResult(
                    timestamp=ts,
                    region=region.name,
                    text=text,
                    matched_keyword=kw,
                    image_path=str(img_path),
                    success=matched,
                )
            )
            if matched:
                append_csv_row(ts, region.name, text, kw)
                self.log.emit(f"[命中] {region.name} → 「{kw}」: {text[:60]}")
            else:
                self.log.emit(f"[扫描] {region.name}: {text[:60] or '(空)'}")

        # 命中才点击
        if any(r.success for r in results) and self._config.click_points:
            self.log.emit(f"[点击] 共 {len(self._config.click_points)} 个点")
            click_results = click_points(self._config.click_points, interval_sec=0.1)
            for name, ok, err in click_results:
                if ok:
                    self.log.emit(f"  ✓ {name}")
                else:
                    self.log.emit(f"  ✗ {name}: {err}")

        return results


class Scheduler(QObject):
    log = pyqtSignal(str)
    status_changed = pyqtSignal(bool)  # True=running

    def __init__(self, config: Config, parent: Optional[QObject] = None):
        super().__init__(parent)
        self._config = config
        self._timer: Optional[QTimer] = None
        self._thread: Optional[QThread] = None
        self._worker: Optional[ScanWorker] = None
        self._stop_event = threading.Event()

    @property
    def is_running(self) -> bool:
        return self._timer is not None and self._timer.isActive()

    def update_config(self, config: Config) -> None:
        self._config = config
        if self._timer is not None:
            self._timer.setInterval(max(1, config.interval_sec) * 1000)

    def start(self) -> None:
        if self.is_running:
            return
        self._stop_event.clear()

        self._thread = QThread()
        self._worker = ScanWorker(self._config, self._stop_event)
        self._worker.moveToThread(self._thread)
        self._worker.log.connect(self._on_log)
        self._worker.finished_round.connect(lambda _: None)  # 留作扩展
        self._thread.start()

        self._timer = QTimer(self)
        self._timer.setInterval(max(1, self._config.interval_sec) * 1000)
        self._timer.timeout.connect(self._worker.run_once)
        self._timer.start()
        self.log.emit(f"[启动] 周期 {self._config.interval_sec} 秒")
        self.status_changed.emit(True)

    def stop(self) -> None:
        if not self.is_running:
            return
        self._stop_event.set()
        if self._timer is not None:
            self._timer.stop()
        # 给 worker 一点时间跑完当前 round
        if self._thread is not None:
            self._thread.quit()
            self._thread.wait(5000)
        self._timer = None
        self._thread = None
        self._worker = None
        self.log.emit("[停止] 已停止监控")
        self.status_changed.emit(False)

    def _on_log(self, msg: str) -> None:
        self.log.emit(msg)
```

```python
from PySide6.QtCore import Qt, QObject, QEvent, Signal
from PySide6.QtGui import QKeyEvent, QMouseEvent, QPainter, QColor, QPen
from PySide6.QtWidgets import QApplication, QWidget


class PickerOverlay(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowFlags(
            Qt.FramelessWindowHint
            | Qt.WindowStaysOnTopHint
            | Qt.Tool
        )
        self.setAttribute(Qt.WA_TranslucentBackground, False)
        self.setCursor(Qt.CrossCursor)
        screen = QApplication.primaryScreen().geometry()
        self.setGeometry(screen)
        self._picked: Optional[Tuple[int, int]] = None

    def paintEvent(self, _event):
        painter = QPainter(self)
        painter.fillRect(self.rect(), QColor(0, 0, 0, 120))
        painter.setPen(QPen(QColor(255, 255, 0), 3))
        font = painter.font()
        font.setPointSize(20)
        painter.setFont(font)
        painter.drawText(self.rect(), Qt.AlignCenter, "单击屏幕拾取坐标 · 按 ESC 取消")

    def mousePressEvent(self, event: QMouseEvent):
        if event.button() == Qt.LeftButton:
            self._picked = (event.globalPosition().x(), event.globalPosition().y())
            self.close()

    def keyPressEvent(self, event: QKeyEvent):
        if event.key() == Qt.Key_Escape:
            self._picked = None
            self.close()

    def picked(self) -> Optional[Tuple[int, int]]:
        return self._picked


def pick_coordinate() -> Optional[Tuple[int, int]]:
    """Block until user clicks (or ESC). Returns (x, y) or None."""
    overlay = PickerOverlay()
    overlay.show()
    # 用本地事件循环，不阻塞主线程
    loop = QApplication.instance()
    while overlay.isVisible():
        loop.processEvents()
    return overlay.picked()
```
